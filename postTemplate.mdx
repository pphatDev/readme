##### 2.2 Javascript
```js
// Config
var barChartOutlineConfig = {
    maintainAspectRatio: false,
    responsive: true,
    scales: {
        y: {
            stacked: true,
            grid: {
                display: true,
                color: "rgba(255,99,132,0.2)"
            }
        },
        x: {
            grid: {
                display: false
            }
        }
    }
};

// Data
var barChartOutlineData = {
    labels: ["Mon", "Tue", "Wen", "Thu", "Fri", "Sat"],
    datasets: [
        {
            label: "New",
            backgroundColor: "rgb(31,168,224,0.2)",
            borderColor: "rgba(31,168,224,1)",
            borderWidth: 2,
            hoverBackgroundColor: "rgba(31,168,224,0.4)",
            hoverBorderColor: "rgba(31,168,224,1)",
            data: [65, 234, 35, 85, 56, 67],
        },
        {
            label: "Contacted",
            backgroundColor: "rgb(92,224,31,0.2)",
            borderColor: "rgba(92,224,31,1)",
            borderWidth: 2,
            hoverBackgroundColor: "rgba(92,224,31,0.4)",
            hoverBorderColor: "rgba(92,224,31,1)",
            data: [65, 100, 35, 81, 56, 55],
            stack: 'combined',
        }
    ]
};

// Create 
new Chart('barChartOutline', {
    type: 'bar',
    options: barChartOutlineConfig,
    data: barChartOutlineData
});
```
